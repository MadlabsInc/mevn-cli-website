MEVN is a scaffolding tool which makes it easy to build universal apps using Mongo, Express, Vue and NodeJS. It minimises the setup time and gets you up to speed using proven technologies.

Getting Started #back to top

A CLI tool for getting started with MEVN stack. It offers a super simple boilerplate project and additional utilities for building a MEVN app. It helps for beginner developers to easily create a template for their Web application development.

Prerequisites

________________

  • Npm - it is a package manager for the JavaScript programming language.
  • Node.js -Node.js is an open-source, cross-platform JavaScript run-time environment that executes JavaScript code outside of a browser.
  • Git - Git is a version control system for tracking changes in computer files and coordinating work on those files among multiple people. It is primarily used for source code management in software development.

Installing

To install the package :- 

npm install -g mevn-cli

Quickstart

 
 npm install -g mevn-cli
 mevn-cli init "Desired File Name"

Index.js

                                            
                                            #!/usr/bin/env node

                                        'use strict'

                                        // Require() includes modules from other files
                                        const program = require('commander');

                                        // Including the modules from the lib 
                                        const init = require('./lib/commands/initlib');
                                        const models = require('./lib/commands/modelslib');
                                        const controllers = require('./lib/commands/controllerslib');
                                        const routes = require('./lib/commands/routeslib');
                                        const config = require('./lib/commands/configlib');
                                        const server = require('./lib/run/server');
                                        const client = require('./lib/run/client');

                                        //Calling the functions from the modules
                                        let initcommand = init.initfunction;
                                        let modelscommand = models.modelsfunction;
                                        let controllerscommand = controllers.controllersfunction;
                                        let routescommand = routes.routesfunction;
                                        let configcommand = config.configfunction;
                                        let runserver = server.serverfunction;
                                        let runclient = client.clientfunction;

                                        // Displays version
                                        program
                                        .version('1.0.0', '-v --version')
                                        .description('A cli tool for MEVN stack.')

                                        //To bring in the project
                                        program
                                        .command('init *appname*')
                                        .description('To init the project')
                                        .action(initcommand);

                                        //To create the controllers file
                                        program
                                        .command('create:controller')
                                        .description('To create controllers-file')
                                        .action(controllerscommand);

                                        // To create router file
                                        program
                                        .command('create:route')
                                        .description('To create router-file')
                                        .action(routescommand);

                                        // To create the configuration file
                                        program
                                        .command('create:config')
                                        .description('To create configuration file for database')
                                        .action(configcommand);

                                        //To create models-file
                                        program
                                        .command('create:model')
                                        .description('To create models-file')
                                        .action(modelscommand);

                                        // To run the client
                                        program
                                        .command('run:client')
                                        .description('To run client')
                                        .action(runclient);

                                        // To run server
                                        program
                                        .command('run:server')
                                        .description('To run server')
                                        .action(runserver);

                                        //For reading the program code 
                                        program.parse(process.argv);

Index.js is the main JavaScript file in which all the modules are called and all the commands are shown and executed. Index.js uses commander for handling command line interface in a more effective manner.

user_controller.js - #back to top

                                        //Importing the user-model from user_scema
                                        import User from '../models/user_schema';
                                        
                                        //Creates the entry using the model 'user" from user_schema
                                        let createData = (req, res, next) => {
                                          
                                          User.create(req.body)
                                          .then( (err) => {
                                            if(err){
                                              return next(err);
                                            }
                                            res.send('Entry created successfully!');  
                                          });
                                        }
                                        
                                        //Finds and sends the required object
                                        let readData = (req, res, next) => {
                                          
                                          User.find( (err, user) => {
                                            if(err){
                                              return next(err);
                                            }
                                            res.send(user); 
                                          });
                                        }
                                        
                                        // Finds and updates the object
                                        let updateData = (req, res, next) => {
                                          
                                          User.findByIdAndUpdate(req.params.id, {$set: req.body}, (err, user) => {
                                            if(err){
                                              return next(err);
                                            }
                                            res.send(user);   
                                          });
                                        }
                                        
                                        //Finds and removes the entry
                                        let deleteData = (req, res, next) => {
                                            
                                          User.findByIdAndRemove(req.params.id, (err) => {
                                            if(err){
                                              return next(err);
                                            }
                                            res.send('Entry deleted successfully!');
                                          });
                                        }
                                        
                                        module.exports = {
                                            createData,
                                            readData,
                                            updateData,
                                            deleteData
                                        };

User_controller.js defines the necessary controllers.
